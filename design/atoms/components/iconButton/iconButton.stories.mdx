<Meta title="Atoms/Icon Buttons" component={IconButton} />

export const Template = (args) => <IconButton {...args} icon={<Icon as={MagnifyingGlass} />} />

## Usage & Guidelines

```ts
import { IconButton, Icon } from '@chakra-ui/react'
import { MagnifyingGlass } from '@phosphor-icons/react'

function SomeComponentWithSecondaryAction() {
    return <IconButton icon={<Icon as={MagnifyingGlass} />}></IconButton>
}
```
